    <h2>Product Calc</h2>

   

    <label>Product Name</label>

    <input type="text" id="prodName" placeholder="e.g. Rack A">

    <label>Quantity</label>

    <input type="number" id="qty" value="1">

    <label>Price per Unit ($)</label>

    <input type="number" id="price" value="100">

    <button onclick="calculate()">Calculate & Save</button>

    <div class="result-box">

        <strong>Total Amount:</strong>

        <div id="total">$0.00</div>

    <h3>Recent History</h3>

    <div id="historyList"></div>

</div>

<script>

    let history = JSON.parse(localStorage.getItem('myHistory')) || [];

    renderHistory();

    function calculate() {

        const name = document.getElementById('prodName').value || "Unnamed Product";

        const q = parseFloat(document.getElementById('qty').value);

        const p = parseFloat(document.getElementById('price').value);

       

        const total = q * p;

        const totalStr = `$${total.toFixed(2)}`;

       

        document.getElementById('total').innerText = totalStr;

        const entry = {

            detail: `${name}: ${q} x $${p}`,

            total: totalStr,

            time: new Date().toLocaleTimeString()

        };

        history.unshift(entry);

        localStorage.setItem('myHistory', JSON.stringify(history));

        renderHistory();

    }

    function renderHistory() {

        const list = document.getElementById('historyList');

        list.innerHTML = history.slice(0, 5).map(item => `

            <div class="history-item">

                <strong>${item.total}</strong> - ${item.detail}<br>

                <small style="color:#999">${item.time}</small>

            </div>

        `).join('');

    }

    if ('serviceWorker' in navigator) {

        navigator.serviceWorker.register('sw.js');

    }

</script>
